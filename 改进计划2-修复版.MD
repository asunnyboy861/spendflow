### 原则3：最小代码原则（已修复）

**原则描述**：优先复用现有模块，抽象需符合"三次法则"（遇到三次再抽），避免过度设计。

**实施标准**：
- ✅ 新功能优先查找现有组件
- ✅ 相同逻辑出现3次才抽象
- ✅ 使用 extension 扩展现有类型
- ✅ 避免创建重复的工具类
- ✅ **强制复用现有组件**：
  - `cardStyle()` - 所有卡片样式
  - `HapticButton` - 所有按钮
  - `EmptyStateView` - 所有空状态
  - `DesignTokens` - 所有间距和字体
  - `Color+Theme` - 所有颜色
  - `Double+Currency` - 所有货币格式化
  - `Date+Extensions` - 所有日期格式化

**代码复用示例**：

### 原则4：重构清理原则

**原则描述**：替换时标记废弃 → 验证无影响 → 删除，仓库提交信息注明清理范围。

**实施流程**：
1. **标记废弃**：使用 `@available(*, deprecated)` 标记
2. **验证测试**：运行所有测试用例，确保无影响
3. **删除代码**：确认无引用后删除
4. **提交说明**：`refactor: 移除废弃的 XXX 模块，已由 YYY 替代`

**示例**：
```swift
// 步骤1：标记废弃
@available(*, deprecated, message: "使用 NewCategoryPicker 替代")

    // ...
}

// 步骤2：验证测试通过后
// 步骤3：删除 OldCategoryPicker.swift
// 步骤4：提交信息
// refactor: 移除废弃的 OldCategoryPicker，已由 NewCategoryPicker 替代
```

## 1.1 银行账户自动同步功能（已修复）

### 1.1.1 需求描述

**用户故事**：
> 作为用户，我希望能够连接我的银行账户，自动导入交易记录，以便减少手动记账的工作量，实现"设置后即忘"的体验。

**功能要点**：
1. 支持连接主流银行（Chase、Bank of America、Wells Fargo等）
2. 自动同步交易记录（每日/手动触发）
3. 支持多账户管理（支票、储蓄、信用卡）
4. 提供同步状态指示器
5. 支持手动刷新和同步历史查看
6. 数据加密存储，保护用户隐私

**非功能需求**：
- 性能：首次同步 < 30秒，增量同步 < 5秒
- 可靠性：同步成功率 > 95%
- 安全性：使用Plaid API，数据加密传输
- 兼容性：iOS 15+
- **后端要求**：需要搭建后端服务处理Plaid回调

**技术实现要求**：
1. **Plaid集成**：
   - 需要申请Plaid开发者账号
   - 需要搭建后端服务（Node.js/Swift on server）
   - 需要处理OAuth回调
   - 需要安全存储access token

2. **替代方案**（初期）：
   - 手动导入CSV
   - 集成第三方同步服务（如Finicity）

#### 1.1.4.1 同步状态模型

**文件**：`Core/Domain/Models/BankSyncStatus.swift`

```swift
import Foundation
import SwiftUI

enum BankSyncStatus: Equatable {
    case notConnected
    case connecting
    case connected(Date)
    case syncing
    case synced(Date)
    case failed(BankSyncError)
    
    var displayText: String {
        switch self {
        case .notConnected:
            return "Not Connected"
        case .connecting:
            return "Connecting..."
        case .connected(let date):
            let formatter = RelativeDateTimeFormatter()
            return "Connected \(formatter.localizedString(for: date, relativeTo: Date()))"
        case .syncing:
            return "Syncing..."
        case .synced(let date):
            let formatter = RelativeDateTimeFormatter()
            return "Synced \(formatter.localizedString(for: date, relativeTo: Date()))"
        case .failed(let error):
            return "Failed: \(error.localizedDescription)"
        }
    }
    
    var icon: String {
        switch self {
        case .notConnected:
            return "link.badge.plus"
        case .connecting, .syncing:
            return "arrow.triangle.2.circlepath"
        case .connected, .synced:
            return "checkmark.link"
        case .failed:
            return "exclamationmark.link"
        }
    }
    
    var color: Color {
        switch self {
        case .notConnected:
            return .secondary
        case .connecting, .syncing:
            return .accentBlue
        case .connected, .synced:
            return .incomeGreen
        case .failed:
            return .expenseRed
        }
    }
}
```

```swift
import Foundation

protocol CategorySuggestionService {
    func suggestCategory(for description: String, amount: Double) -> CategorySuggestion
    func suggestCategories(for description: String, amount: Double, limit: Int) -> [CategorySuggestion]
    func learnFromAdjustment(original: CategorySuggestion, corrected: Category)
}

class CompositeCategorySuggestionService: CategorySuggestionService {
    
    private let ruleBasedSuggester: RuleBasedSuggester
    private let learningService: CategoryLearningService
    
    init(
        ruleBasedSuggester: RuleBasedSuggester,
        learningService: CategoryLearningService
    ) {
        self.ruleBasedSuggester = ruleBasedSuggester
        self.learningService = learningService
    }
    
    func suggestCategory(for description: String, amount: Double) -> CategorySuggestion {
        // 先检查历史学习
        if let learned = learningService.getSuggestion(for: description) {
            return learned
        }
        
        // 再使用规则
        let ruleSuggestions = ruleBasedSuggester.suggestCategories(for: description, amount: amount, limit: 1)
        if let suggestion = ruleSuggestions.first {
            return suggestion
        }
        
        // 返回默认分类
        return CategorySuggestion(
            category: Category.commonCategories.last!,
            confidence: 0.3,
            source: .defaultCategory
        )
    }
    
    func suggestCategories(for description: String, amount: Double, limit: Int) -> [CategorySuggestion] {
        var suggestions: [CategorySuggestion] = []
        
        // 合并历史学习和规则建议
        if let learned = learningService.getSuggestion(for: description) {
            suggestions.append(learned)
        }
        
        let ruleSuggestions = ruleBasedSuggester.suggestCategories(for: description, amount: amount, limit: limit)
        suggestions.append(contentsOf: ruleSuggestions)
        
        // 去重并排序
        let uniqueSuggestions = Dictionary(grouping: suggestions, by: { $0.category.id })
            .values
            .compactMap { $0.first }
            .sorted { $0.confidence > $1.confidence }
        
        return Array(uniqueSuggestions.prefix(limit))
    }
    
    func learnFromAdjustment(original: CategorySuggestion, corrected: Category) {
        learningService.learn(
            category: corrected,
            for: original.category.name
        )
    }
}
```

```swift
import Foundation

struct BudgetAllocation: Identifiable {
    let id = UUID()
    let category: String
    var allocated: Double
    var spent: Double
    var remaining: Double {
        allocated - spent
    }
    var percentage: Double {
        guard allocated > 0 else { return 0 }
        return spent / allocated * 100
    }
}

struct AgeOfMoney: Equatable {
    let days: Int
    let trend: AgeOfMoneyTrend
    
    enum AgeOfMoneyTrend {
        case increasing
        case stable
        case decreasing
    }
    
    var displayText: String {
        "\(days) days"
    }
    
    var description: String {
        switch trend {
        case .increasing:
            return "Your money is staying in your account longer. Great job!"
        case .stable:
            return "Your spending pattern is consistent."
        case .decreasing:
            return "Consider reducing expenses to increase your buffer."
        }
    }
}
```

#### 2.1.3.2 零基预算视图模型（已修复）

**文件**：`Core/Presentation/ViewModels/ZeroBasedBudgetViewModel.swift`

```swift
import Combine
import Foundation

class ZeroBasedBudgetViewModel: ObservableObject {
    @Published var allocations: [BudgetAllocation] = []
    @Published var totalIncome: Double = 0
    @Published var totalAllocated: Double = 0
    @Published var ageOfMoney: AgeOfMoney?
    @Published var isLoading: Bool = false
    
    private let transactionRepository: TransactionRepository
    private let budgetRepository: BudgetRepository
    private let ageOfMoneyCalculator: AgeOfMoneyCalculator
    private var cancellables = Set<AnyCancellable>()
    
    init(
        transactionRepository: TransactionRepository,
        budgetRepository: BudgetRepository,
        ageOfMoneyCalculator: AgeOfMoneyCalculator
    ) {
        self.transactionRepository = transactionRepository
        self.budgetRepository = budgetRepository
        self.ageOfMoneyCalculator = ageOfMoneyCalculator
        
        loadData()
    }
    
    func loadData() {
        Task {
            await MainActor.run {
                self.isLoading = true
            }
            
            let transactions = transactionRepository.fetchAll()
            let budgets = budgetRepository.fetchAll()
            
            // 计算收入
            let income = transactions
                .filter { $0.type == .income }
                .map { $0.amount }
                .reduce(0, +)
            
            // 计算分配
            let allocations = budgets.map { budget in
                let spent = transactions
                    .filter { $0.category == budget.category }
                    .map { $0.amount }
                    .reduce(0, +)
                
                return BudgetAllocation(
                    category: budget.category,
                    allocated: budget.amount,
                    spent: spent
                )
            }
            
            // 计算资金年龄
            let ageOfMoney = ageOfMoneyCalculator.calculate(from: transactions)
            
            await MainActor.run {
                self.totalIncome = income
                self.allocations = allocations
                self.totalAllocated = allocations.map { $0.allocated }.reduce(0, +)
                self.ageOfMoney = ageOfMoney
                self.isLoading = false
            }
        }
    }
    
    func allocateIncome(amount: Double, to category: String) {
        // ✅ 通过Repository更新预算
        Task {
            if let budget = budgetRepository.fetchAll().first(where: { $0.category == category }) {
                var updatedBudget = budget
                updatedBudget.amount += amount
                try budgetRepository.update(updatedBudget)
                loadData()
            }
        }
    }
    
    func transferBudget(from: String, to: String, amount: Double) {
        Task {
            let budgets = budgetRepository.fetchAll()
            
            if let fromBudget = budgets.first(where: { $0.category == from }),
               let toBudget = budgets.first(where: { $0.category == to }) {
                var updatedFromBudget = fromBudget
                updatedFromBudget.amount -= amount
                try budgetRepository.update(updatedFromBudget)
                
                var updatedToBudget = toBudget
                updatedToBudget.amount += amount
                try budgetRepository.update(updatedToBudget)
                
                loadData()
            }
        }
    }
}
```

```swift
// ✅ 统一同步管理
protocol SyncCoordinator {
    var iCloudSync: SyncService { get }
    var bankSync: BankSyncService? { get }
    
    func syncAll() async throws
}

class DefaultSyncCoordinator: SyncCoordinator {
    let iCloudSync: SyncService
    let bankSync: BankSyncService?
    
    init(iCloudSync: SyncService, bankSync: BankSyncService? = nil) {
        self.iCloudSync = iCloudSync
        self.bankSync = bankSync
    }
    
    func syncAll() async throws {
        // 先同步银行数据
        if let bankSync = bankSync {
            try await bankSync.syncAllAccounts()
        }
        
        // 再同步到iCloud
        try await iCloudSync.syncNow()
    }
}
```
