### 原则3：最小代码原则

**原则描述**：优先复用现有模块，抽象需符合"三次法则"（遇到三次再抽），避免过度设计。

**实施标准**：
- ✅ 新功能优先查找现有组件
- ✅ 相同逻辑出现3次才抽象
- ✅ 使用 extension 扩展现有类型
- ✅ 避免创建重复的工具类

**代码复用示例**：
```swift
// ❌ 错误：创建新的卡片组件
struct BankAccountCard: View {
    var body: some View {
        VStack { /* 自定义样式 */ }
            .background(Color.white)
            .cornerRadius(12)
    }
}

// ✅ 正确：复用现有的 cardStyle()
struct BankAccountCard: View {
    var body: some View {
        VStack { /* 内容 */ }
            .cardStyle()  // 复用现有样式
    }
}
```

### 原则4：重构清理原则

**原则描述**：替换时标记废弃 → 验证无影响 → 删除，仓库提交信息注明清理范围。

**实施流程**：
1. **标记废弃**：使用 `@available(*, deprecated)` 标记
2. **验证测试**：运行所有测试用例，确保无影响
3. **删除代码**：确认无引用后删除
4. **提交说明**：`refactor: 移除废弃的 XXX 模块，已由 YYY 替代`

**示例**：
```swift
// 步骤1：标记废弃
@available(*, deprecated, message: "使用 NewCategoryPicker 替代")
struct OldCategoryPicker: View {
    // ...
}

// 步骤2：验证测试通过后
// 步骤3：删除 OldCategoryPicker.swift
// 步骤4：提交信息
// refactor: 移除废弃的 OldCategoryPicker，已由 NewCategoryPicker 替代
```

## 1.1 银行账户自动同步功能

### 1.1.1 需求描述

**用户故事**：
> 作为用户，我希望能够连接我的银行账户，自动导入交易记录，以便减少手动记账的工作量，实现"设置后即忘"的体验。

**功能要点**：
1. 支持连接主流银行（Chase、Bank of America、Wells Fargo等）
2. 自动同步交易记录（每日/手动触发）
3. 支持多账户管理（支票、储蓄、信用卡）
4. 提供同步状态指示器
5. 支持手动刷新和同步历史查看
6. 数据加密存储，保护用户隐私

**非功能需求**：
- 性能：首次同步 < 30秒，增量同步 < 5秒
- 可靠性：同步成功率 > 95%
- 安全性：使用Plaid API，数据加密传输
- 兼容性：iOS 15+，支持美国主流银行

**美国用户期望**：
- 根据NerdWallet调研，87%的美国用户期望预算APP能自动同步银行账户
- Mint、Monarch、Copilot等主流APP都支持此功能
- 这是SpendFlow与主流APP最大的差距

#### 1.1.4.4 Plaid服务实现

**文件**：`Features/BankSync/Services/PlaidService.swift`

```swift
import Foundation
import Combine
import LinkKit

class PlaidService: BankSyncService {
    
    private let transactionRepository: TransactionRepository
    private let transactionImporter: TransactionImporter
    private let syncStatusSubject = CurrentValueSubject<BankSyncStatus, Never>(.notConnected)
    private let connectedAccountsSubject = CurrentValueSubject<[BankAccount], Never>([])
    private var cancellables = Set<AnyCancellable>()
    
    var syncStatusPublisher: AnyPublisher<BankSyncStatus, Never> {
        syncStatusSubject.eraseToAnyPublisher()
    }
    
    var connectedAccountsPublisher: AnyPublisher<[BankAccount], Never> {
        connectedAccountsSubject.eraseToAnyPublisher()
    }
    
    init(
        transactionRepository: TransactionRepository,
        transactionImporter: TransactionImporter
    ) {
        self.transactionRepository = transactionRepository
        self.transactionImporter = transactionImporter
    }
    
    func connectBank() async throws -> String {
        syncStatusSubject.send(.connecting)
        
        return try await withCheckedThrowingContinuation { continuation in
            let configuration = PLKConfiguration(
                key: AppConstants.plaidClientId,
                env: .sandbox,
                products: [.transactions]
            )
            
            let linkViewController = PLKPlaidLinkViewController(configuration: configuration, delegate: self)
            
            linkViewController.resultPublisher
                .sink { result in
                    switch result {
                    case .success(let publicToken):
                        continuation.resume(returning: publicToken)
                    case .failure(let error):
                        continuation.resume(throwing: BankSyncError.connectionFailed(error.localizedDescription))
                    }
                }
                .store(in: &self.cancellables)
            
            linkViewController.start()
        }
    }
    
    func completeConnection(publicToken: String) async throws {
        syncStatusSubject.send(.connecting)
        
        do {
            let accounts = try await exchangePublicToken(publicToken)
            connectedAccountsSubject.send(accounts)
            syncStatusSubject.send(.connected(Date()))
        } catch {
            syncStatusSubject.send(.failed(.connectionFailed(error.localizedDescription)))
            throw error
        }
    }
    
    func syncTransactions(for accountId: UUID) async throws {
        syncStatusSubject.send(.syncing)
        
        guard let account = connectedAccountsSubject.value.first(where: { $0.id == accountId }) else {
            throw BankSyncError.accountNotFound
        }
        
        do {
            let transactions = try await fetchTransactions(for: account.plaidAccountId)
            try await transactionImporter.importTransactions(transactions, to: accountId)
            
            syncStatusSubject.send(.synced(Date()))
        } catch {
            syncStatusSubject.send(.failed(.syncFailed(error.localizedDescription)))
            throw error
        }
    }
    
    func syncAllAccounts() async throws {
        syncStatusSubject.send(.syncing)
        
        for account in connectedAccountsSubject.value {
            try await syncTransactions(for: account.id)
        }
        
        syncStatusSubject.send(.synced(Date()))
    }
    
    func disconnectAccount(_ accountId: UUID) async throws {
        var accounts = connectedAccountsSubject.value
        accounts.removeAll { $0.id == accountId }
        connectedAccountsSubject.send(accounts)
        
        if accounts.isEmpty {
            syncStatusSubject.send(.notConnected)
        }
    }
    
    func refreshAccount(_ accountId: UUID) async throws {
        try await syncTransactions(for: accountId)
    }

    }
}
```

```swift
        categorySuggester: CategorySuggester
    ) {
        self.transactionRepository = transactionRepository
        self.categorySuggester = categorySuggester
    }
    
    func importTransactions(
        _ plaidTransactions: [PlaidTransaction],
        to accountId: UUID
    ) async throws {
        for plaidTransaction in plaidTransactions {
            let suggestedCategory = categorySuggester.suggestCategory(
                for: plaidTransaction.name,
                amount: plaidTransaction.amount
            )
            
            let transaction = Transaction(
                amount: abs(plaidTransaction.amount),
                category: suggestedCategory.name,
                date: plaidTransaction.date,
                note: plaidTransaction.merchantName ?? plaidTransaction.name,
                type: plaidTransaction.amount < 0 ? .income : .expense
            )
            
            try transactionRepository.save(transaction)
        }
    }
```

```swift
import Foundation

protocol CategorySuggester {
    func suggestCategory(for description: String, amount: Double) -> CategorySuggestion
    func suggestCategories(for description: String, amount: Double, limit: Int) -> [CategorySuggestion]
    func learnFromUserChoice(description: String, amount: Double, category: Category)
}

class RuleBasedSuggester: CategorySuggester {
    
    private let transactionRepository: TransactionRepository
    private var categoryRules: [CategoryRule] = []
    
    init(transactionRepository: TransactionRepository) {
        self.transactionRepository = transactionRepository
        loadRules()
    }
    
    func suggestCategory(for description: String, amount: Double) -> CategorySuggestion {
        let suggestions = suggestCategories(for: description, amount: amount, limit: 1)
        return suggestions.first ?? CategorySuggestion(
            category: Category.commonCategories.last!,  // "Other"
            confidence: 0.3,
            source: .defaultCategory
        )
    }
    
    func suggestCategories(for description: String, amount: Double, limit: Int) -> [CategorySuggestion] {
        var suggestions: [CategorySuggestion] = []
        
        // 1. 检查用户历史
        let historySuggestion = checkUserHistory(description: description)
        if let suggestion = historySuggestion {
            suggestions.append(suggestion)
        }
        
        // 2. 检查规则
        let ruleSuggestions = checkRules(description: description, amount: amount)
        suggestions.append(contentsOf: ruleSuggestions)
        
        // 3. 检查关键词
        let keywordSuggestions = checkKeywords(description: description)
        suggestions.append(contentsOf: keywordSuggestions)
        
        // 去重并排序
        let uniqueSuggestions = Dictionary(grouping: suggestions, by: { $0.category.id })
            .values
            .compactMap { $0.first }
            .sorted { $0.confidence > $1.confidence }
        
        return Array(uniqueSuggestions.prefix(limit))
    }
    
    func learnFromUserChoice(description: String, amount: Double, category: Category) {
        let rule = CategoryRule(
            keywords: extractKeywords(from: description),
            category: category,
            confidence: 1.0
        )
        categoryRules.append(rule)
        saveRules()
    }
    
    private func checkUserHistory(description: String) -> CategorySuggestion? {
        let transactions = transactionRepository.fetchAll()
        let similarTransactions = transactions.filter {
            $0.note?.lowercased().contains(description.lowercased()) ?? false
        }
        
        guard let mostCommon = similarTransactions.mostCommonCategory() else {
            return nil
        }
        
        let confidence = Double(similarTransactions.count) / Double(max(transactions.count, 1))
        
        return CategorySuggestion(
            category: mostCommon,
            confidence: min(confidence + 0.5, 0.95),
            source: .userHistory
        )
    }
    
    private func checkRules(description: String, amount: Double) -> [CategorySuggestion] {
        var suggestions: [CategorySuggestion] = []
        
        for rule in categoryRules {
            let matchCount = rule.keywords.filter { description.lowercased().contains($0.lowercased()) }.count
            if matchCount > 0 {
                let confidence = Double(matchCount) / Double(rule.keywords.count) * rule.confidence
                suggestions.append(CategorySuggestion(
                    category: rule.category,
                    confidence: confidence,
                    source: .rule
                ))
            }
        }
        
        return suggestions
    }
    
    private func checkKeywords(description: String) -> [CategorySuggestion] {
        let keywordMap: [String: (Category, Double)] = [
            "grocery": (Category.commonCategories[0], 0.85),  // Food & Dining
            "restaurant": (Category.commonCategories[0], 0.90),
            "uber": (Category.commonCategories[1], 0.90),  // Transportation
            "gas": (Category.commonCategories[1], 0.85),
            "amazon": (Category.commonCategories[2], 0.80),  // Shopping
            "netflix": (Category.commonCategories[8], 0.95),  // Subscription
            "spotify": (Category.commonCategories[8], 0.95),
            "salary": (Category.incomeCategories[0], 0.95),  // Salary
            "payroll": (Category.incomeCategories[0], 0.95)
        ]
        
        var suggestions: [CategorySuggestion] = []
        
        for (keyword, (category, confidence)) in keywordMap {
            if description.lowercased().contains(keyword) {
                suggestions.append(CategorySuggestion(
                    category: category,
                    confidence: confidence,
                    source: .rule
                ))
            }
        }
        
        return suggestions
    }
    
    private func extractKeywords(from text: String) -> [String] {
        let stopWords = ["the", "a", "an", "and", "or", "but", "in", "on", "at", "to", "for"]
        return text.lowercased()
            .components(separatedBy: .whitespacesAndNewlines)
            .filter { !$0.isEmpty && !stopWords.contains($0) }
    }
    
    private func loadRules() {
        // 从UserDefaults或CoreData加载规则
    }
    
    private func saveRules() {

    }
}

extension Array where Element == Transaction {
    func mostCommonCategory() -> Category? {
        let categoryCounts = self.reduce(into: [Category: Int]()) { counts, transaction in
            if let category = Category.commonCategories.first(where: { $0.name == transaction.category }) {
```

#### 2.2.3.1 账单模型

**文件**：`Features/Bills/Models/Bill.swift`

```swift
import Foundation

struct Bill: Identifiable, Codable {
    let id: UUID
    var name: String
    var amount: Double
    var dueDate: Date
    var frequency: BillFrequency
    var category: String
    var reminderDays: Int
    var isActive: Bool
    var lastPaidDate: Date?
    
    init(
        id: UUID = UUID(),
        name: String,
        amount: Double,
        dueDate: Date,
        frequency: BillFrequency,
        category: String,
        reminderDays: Int = 3,
        isActive: Bool = true,
        lastPaidDate: Date? = nil
    ) {
        self.id = id
        self.name = name
        self.amount = amount
        self.dueDate = dueDate
        self.frequency = frequency
        self.category = category
        self.reminderDays = reminderDays
        self.isActive = isActive
        self.lastPaidDate = lastPaidDate
    }
    
    var nextDueDate: Date {
        let calendar = Calendar.current
        var nextDate = dueDate
        
        while nextDate < Date() {
            switch frequency {
            case .monthly:
                nextDate = calendar.date(byAdding: .month, value: 1, to: nextDate) ?? nextDate
            case .quarterly:
                nextDate = calendar.date(byAdding: .month, value: 3, to: nextDate) ?? nextDate
            case .yearly:
                nextDate = calendar.date(byAdding: .year, value: 1, to: nextDate) ?? nextDate
            case .oneTime:
                return dueDate
            }
        }
        
        return nextDate
    }
    
    var daysUntilDue: Int {
        let calendar = Calendar.current
        let components = calendar.dateComponents([.day], from: Date(), to: nextDueDate)
        return components.day ?? 0
    }
    
    var isOverdue: Bool {
        daysUntilDue < 0
    }
    
    var isDueSoon: Bool {
        daysUntilDue <= reminderDays && daysUntilDue >= 0
    }
    
    var status: BillStatus {
        if isOverdue {
            return .overdue
        } else if isDueSoon {
            return .dueSoon
        } else {
            return .upcoming
        }
    }
}

enum BillFrequency: String, Codable, CaseIterable {
    case monthly = "Monthly"
    case quarterly = "Quarterly"
    case yearly = "Yearly"
    case oneTime = "One Time"
}

enum BillStatus {
    case overdue
    case dueSoon
    case upcoming
    
    var color: String {
        switch self {
        case .overdue: return "FF3B30"
        case .dueSoon: return "FF9500"
        case .upcoming: return "34C759"
        }
    }
    
    var icon: String {
        switch self {
        case .overdue: return "exclamationmark.circle.fill"
        case .dueSoon: return "clock.fill"
        case .upcoming: return "checkmark.circle.fill"
        }
    }
}
```

```swift
import Foundation
import UserNotifications

class BillReminderService {
    
    private let notificationService: NotificationService
    
    init(notificationService: NotificationService) {
        self.notificationService = notificationService
    }
    
    func scheduleReminder(for bill: Bill) async throws {
        let reminderDate = Calendar.current.date(
            byAdding: .day,
            value: -bill.reminderDays,
            to: bill.nextDueDate
        ) ?? bill.nextDueDate
        
        let content = UNMutableNotificationContent()
        content.title = "Bill Reminder"
        content.body = "\(bill.name) is due in \(bill.reminderDays) days. Amount: \(bill.amount.currencyFormatted)"
        content.sound = .default
        content.badge = 1
        
        let triggerDate = Calendar.current.dateComponents(
            [.year, .month, .day, .hour, .minute],
            from: reminderDate
        )
        
        let trigger = UNCalendarNotificationTrigger(
            dateMatching: triggerDate,
            repeats: false
        )
        
        let request = UNNotificationRequest(
            identifier: bill.id.uuidString,
            content: content,
            trigger: trigger
        )
        
        try await notificationService.schedule(request)
    }
    
    func cancelReminder(for bill: Bill) async throws {
        try await notificationService.cancel(identifier: bill.id.uuidString)
    }
    
    func updateReminder(for bill: Bill) async throws {
        try await cancelReminder(for: bill)
        try await scheduleReminder(for: bill)
    }
}
```

```swift
import Foundation

struct BudgetAllocation: Identifiable, Codable {
    let id: UUID
    let category: Category
    var allocated: Double
    var spent: Double
    var available: Double
    
    init(
        id: UUID = UUID(),
        category: Category,
        allocated: Double = 0,
        spent: Double = 0
    ) {
        self.id = id
        self.category = category
        self.allocated = allocated
        self.spent = spent
        self.available = allocated - spent
    }
    
    mutating func allocate(_ amount: Double) {
        allocated += amount
        available = allocated - spent
    }
    
    mutating func spend(_ amount: Double) {
        spent += amount
        available = allocated - spent
    }
    
    mutating func transfer(to other: inout BudgetAllocation, amount: Double) {
        guard available >= amount else { return }
        allocated -= amount
        other.allocated += amount
        available = allocated - spent
        other.available = other.allocated - other.spent
    }
    
    var progress: Double {
        guard allocated > 0 else { return 0 }
        return spent / allocated
    }
    
    var isOverBudget: Bool {
        available < 0
    }
}
```

```swift
import Foundation

class AgeOfMoneyCalculator {
    
    private let transactionRepository: TransactionRepository
    
    init(transactionRepository: TransactionRepository) {
        self.transactionRepository = transactionRepository
    }
    
    func calculateAgeOfMoney() -> Int {
        let transactions = transactionRepository.fetchAll()
            .sorted { $0.date < $1.date }
        
        var incomeQueue: [(date: Date, amount: Double)] = []
        var totalAge: Double = 0
        var totalSpent: Double = 0
        
        for transaction in transactions {
            if transaction.type == .income {
                incomeQueue.append((date: transaction.date, amount: transaction.amount))
            } else {
                var remaining = transaction.amount
                totalSpent += transaction.amount
                
                while remaining > 0 && !incomeQueue.isEmpty {
                    var income = incomeQueue.removeFirst()
                    let used = min(remaining, income.amount)
                    
                    let age = Calendar.current.dateComponents([.day], from: income.date, to: transaction.date).day ?? 0
                    totalAge += Double(age) * (used / transaction.amount)
                    
                    remaining -= used
                    income.amount -= used
                    
                    if income.amount > 0 {
                        incomeQueue.insert(income, at: 0)
                    }
                }
            }
        }
        
        return totalSpent > 0 ? Int(totalAge) : 0
    }
}
```
