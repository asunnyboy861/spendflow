## 一、项目概述

### 1.1 项目背景

基于深入的市场调研和用户痛点分析，我们发现现有预算管理APP存在以下核心问题：

1. **信息过载**：用户看到的不是"我还能花多少钱"，而是复杂的图表和报表
2. **输入繁琐**：手动记录每一笔交易让人沮丧，用户很快放弃使用
3. **缺乏实时性**：用户打开APP想知道"现在的情况"，而不是"上个月的情况"
4. **多账户混乱**：用户有多个银行账户、信用卡，现有APP无法聚合显示

### 1.2 核心价值主张

**SpendFlow的核心使命**：让用户一眼看到"我还能花多少钱"

**差异化定位**：
- ❌ 不做：复杂的财务报表、详细的账户管理、投资分析
- ✅ 专注：剩余可用预算的实时聚合显示

### 1.3 目标用户

**主要用户画像**：

1. **年轻专业人士（25-35岁）**
   - 收入稳定，有一定储蓄意识
   - 生活忙碌，追求效率
   - 习惯使用移动应用管理生活
   - 愿意为简化生活的工具付费

2. **注重简洁的用户**
   - 讨厌复杂的财务软件
   - 只想知道"底线数字"
   - 不想学习新系统
   - 希望低摩擦体验

3. **多账户用户**
   - 有2-3个银行账户
   - 使用多张信用卡
   - 希望看到整体财务状况
   - 避免在多个APP间切换

### 4.3 核心功能实现代码

#### 功能1：剩余可用实时计算引擎

```swift
import Foundation
import Combine

// MARK: - 核心计算引擎
class BudgetCalculator: ObservableObject {
    // MARK: - Published Properties
    @Published var remainingToday: Double = 0
    @Published var remainingThisWeek: Double = 0
    @Published var remainingThisMonth: Double = 0
    
    @Published var spentToday: Double = 0
    @Published var spentThisWeek: Double = 0
    @Published var spentThisMonth: Double = 0
    
    @Published var budgetProgress: Double = 0  // 0.0 - 1.0
    
    // MARK: - Dependencies
    private let transactionRepository: TransactionRepository
    private let budgetRepository: BudgetRepository
    private var cancellables = Set<AnyCancellable>()
    
    // MARK: - Initialization
    init(transactionRepository: TransactionRepository, 
         budgetRepository: BudgetRepository) {
        self.transactionRepository = transactionRepository
        self.budgetRepository = budgetRepository
        
        setupObservers()
    }
    
    // MARK: - Setup
    private func setupObservers() {
        // 监听交易变化，实时更新剩余金额
        transactionRepository.transactionsPublisher
            .receive(on: DispatchQueue.main)
            .sink { [weak self] _ in
                self?.recalculateAll()
            }
            .store(in: &cancellables)
    }
    
    // MARK: - Core Calculation Methods
    func recalculateAll() {
        calculateRemainingToday()
        calculateRemainingThisWeek()
        calculateRemainingThisMonth()
    }
    
    /// 计算今日剩余可用
    func calculateRemainingToday() {
        let calendar = Calendar.current
        let today = calendar.startOfDay(for: Date())
        
        // 获取今日支出
        let todayExpenses = transactionRepository.fetchTransactions(
            from: today,
            to: calendar.date(byAdding: .day, value: 1, to: today)!
        ).filter { $0.type == .expense }
         .reduce(0) { $0 + $1.amount }
        
        // 获取每日预算
        let dailyBudget = budgetRepository.fetchBudget(for: .daily)?.amount ?? 0
        
        // 计算剩余
        spentToday = todayExpenses
        remainingToday = max(0, dailyBudget - todayExpenses)
    }
    
    /// 计算本周剩余可用
    func calculateRemainingThisWeek() {
        let calendar = Calendar.current
        let weekStart = calendar.date(from: calendar.dateComponents([.yearForWeekOfYear, .weekOfYear], from: Date()))!
        let weekEnd = calendar.date(byAdding: .day, value: 7, to: weekStart)!
        
        // 获取本周支出
        let weekExpenses = transactionRepository.fetchTransactions(
            from: weekStart,
            to: weekEnd
        ).filter { $0.type == .expense }
         .reduce(0) { $0 + $1.amount }
        
        // 获取每周预算
        let weeklyBudget = budgetRepository.fetchBudget(for: .weekly)?.amount ?? 0
        
        // 计算剩余
        spentThisWeek = weekExpenses
        remainingThisWeek = max(0, weeklyBudget - weekExpenses)
    }
    
    /// 计算本月剩余可用
    func calculateRemainingThisMonth() {
        let calendar = Calendar.current
        let monthStart = calendar.date(from: calendar.dateComponents([.year, .month], from: Date()))!
        let monthEnd = calendar.date(byAdding: .month, value: 1, to: monthStart)!
        
        // 获取本月支出
        let monthExpenses = transactionRepository.fetchTransactions(
            from: monthStart,
            to: monthEnd
        ).filter { $0.type == .expense }
         .reduce(0) { $0 + $1.amount }
        
        // 获取每月预算
        let monthlyBudget = budgetRepository.fetchBudget(for: .monthly)?.amount ?? 0
        
        // 计算剩余和进度
        spentThisMonth = monthExpenses
        remainingThisMonth = max(0, monthlyBudget - monthExpenses)
        budgetProgress = monthlyBudget > 0 ? spentThisMonth / monthlyBudget : 0
    }
    
    /// 智能预测：基于当前消费速度预测月底情况
    func predictEndOfMonth() -> BudgetPrediction {
        let calendar = Calendar.current
        let monthStart = calendar.date(from: calendar.dateComponents([.year, .month], from: Date()))!
        let monthEnd = calendar.date(byAdding: .month, value: 1, to: monthStart)!
        
        let daysPassed = calendar.dateComponents([.day], from: monthStart, to: Date()).day! + 1
        let totalDays = calendar.dateComponents([.day], from: monthStart, to: monthEnd).day!
        let daysRemaining = totalDays - daysPassed
        
        // 计算日均消费
        let dailyAverage = daysPassed > 0 ? spentThisMonth / Double(daysPassed) : 0
        
        // 预测月底总支出
        let predictedTotalSpent = spentThisMonth + (dailyAverage * Double(daysRemaining))
        
        // 预测剩余
        let monthlyBudget = budgetRepository.fetchBudget(for: .monthly)?.amount ?? 0
        let predictedRemaining = monthlyBudget - predictedTotalSpent
        
        return BudgetPrediction(
            predictedTotalSpent: predictedTotalSpent,
            predictedRemaining: predictedRemaining,
            dailyAverage: dailyAverage,
            isOnTrack: predictedTotalSpent <= monthlyBudget
        )
    }
}

    let predictedTotalSpent: Double
    let predictedRemaining: Double
    let dailyAverage: Double
    let isOnTrack: Bool
}
```

```swift
import SwiftUI

struct RemainingCard: View {
    let remaining: Double
    let budget: Double
    let spent: Double
    let period: TimePeriod
    
    @Environment(\.colorScheme) var colorScheme
    
    var body: some View {

                .font(.caption)
                .foregroundColor(.textSecondary)
                .textCase(.uppercase)
                .tracking(1.5)
            
            // 剩余金额（大数字）
            Text("$\(remaining, specifier: "%.2f")")
                .font(.remainingAmount)
                .foregroundColor(colorForRemaining)
                .contentTransition(.numericText())
            
            // 进度条
            GeometryReader { geometry in
                ZStack(alignment: .leading) {
                    // 背景轨道
                    RoundedRectangle(cornerRadius: 8)
                        .fill(Color.gray.opacity(0.2))
                        .frame(height: 12)
                    
                    // 进度条
                    RoundedRectangle(cornerRadius: 8)
                        .fill(
                            LinearGradient(
                                colors: [colorForRemaining, colorForRemaining.opacity(0.7)],
                                startPoint: .leading,
                                endPoint: .trailing
                            )
                        )
                        .frame(width: geometry.size.width * progressPercentage, height: 12)
                        .animation(.spring(response: 0.6, dampingFraction: 0.7), value: spent)
                }
            }
            .frame(height: 12)
            
            // 进度文字
            HStack {
                Text("$\(spent, specifier: "%.0f") spent")
                    .font(.bodySmall)
                    .foregroundColor(.textSecondary)
                
                Spacer()
                
                Text("\(Int(progressPercentage * 100))% used")
                    .font(.bodySmall)
                    .fontWeight(.semibold)
                    .foregroundColor(colorForRemaining)
            }
        }
        .padding(24)
        .background(
            RoundedRectangle(cornerRadius: 20)
                .fill(Color.backgroundSecondary)
                .shadow(
                    color: colorScheme == .dark ? Color.clear : Color.black.opacity(0.08),
                    radius: 12,
                    x: 0,
                    y: 4
                )
        )
    }
    
    // MARK: - Computed Properties
    private var titleText: String {
        switch period {
        case .daily: return "You Can Spend Today"
        case .weekly: return "You Can Spend This Week"
        case .monthly: return "You Can Spend This Month"
        }
    }
    
    private var progressPercentage: Double {
        guard budget > 0 else { return 0 }
        return min(spent / budget, 1.0)
    }
    
    private var colorForRemaining: Color {
        let percentage = progressPercentage
        if percentage < 0.7 { return .budgetSafe }
        if percentage < 0.9 { return .budgetWarning }
        return .budgetDanger
    }
}
```
